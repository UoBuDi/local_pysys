### 2.8 结果处理逻辑对比

#### 收费路径匹配系统实现 (Node.js)

```javascript
// server/index.js:226-282
const processedResults = results.map((item, index) => {
  const unitNames = item['拆分收费单元名称组合'].split('|');
  const amounts = item['拆分收费金额组合'].split('|');
  
  let filteredUnits = [];
  let filteredAmounts = [];
  
  if (excludeUnits) {
    const excludeUnitsArray = excludeUnits.split(',').map(item => item.trim());
    for (let i = 0; i < unitNames.length; i++) {
      const unitName = unitNames[i].trim();
      const amount = amounts[i].trim();
      
      const exclude = excludeUnitsArray.some(excludeUnit => 
        unitName.includes(excludeUnit)
      );
      
      if (!exclude) {
        filteredUnits.push(unitName);
        filteredAmounts.push(amount);
      }
    }
  } else {
    filteredUnits = unitNames.map(item => item.trim());
    filteredAmounts = amounts.map(item => item.trim());
  }
  
  const totalAmount = filteredAmounts.reduce((sum, amount) => {
    return sum + (parseFloat(amount) || 0);
  }, 0);
  
  const path = filteredUnits.join(' → ');
  
  const result = {
    id: index + 1,
    path: path,
    totalAmount: totalAmount.toFixed(2),
    startTime: item['计费交易起点时间'],
    endTime: item['计费交易终点时间'],
    exitName: item['收费出口名称']
  };
  
  for (const key in item) {
    if (!result.hasOwnProperty(key)) {
      result[key] = item[key];
    }
  }
  
  return result;
});
```

#### 当前项目实现 (Python)

```python
# backend/main.py:2063-2108
processedResults = []
for index, item in enumerate(results):
    unitNames = str(item.get('拆分收费单元名称组合', '')).split('|')
    amounts = str(item.get('拆分收费金额组合', '')).split('|')
    
    filteredUnits = []
    filteredAmounts = []
    
    if excludeUnits:
        excludeUnitsArray = [item.strip() for item in excludeUnits.split(',')]
        for i in range(min(len(unitNames), len(amounts))):
            unitName = unitNames[i].strip()
            amount = amounts[i].strip()
            
            exclude = any(excludeUnit in unitName for excludeUnit in excludeUnitsArray if excludeUnit)
            
            if not exclude:
                filteredUnits.append(unitName)
                filteredAmounts.append(amount)
    else:
        filteredUnits = [unitName.strip() for unitName in unitNames]
        filteredAmounts = [amount.strip() for amount in amounts]
    
    totalAmount = 0.0
    for amount in filteredAmounts:
        try:
            totalAmount += float(amount)
        except ValueError:
            pass
    
    path = ' → '.join(filteredUnits)
    
    result = {
        'id': index + 1,
        'path': path,
        'totalAmount': f"{totalAmount:.2f}",

        'endTime': item.get('计费交易终点时间', ''),
        'exitName': item.get('收费出口名称', '')
    }
    
    for key, value in item.items():
        if key not in result:
            result[key] = value
    
    processedResults.append(result)
```
