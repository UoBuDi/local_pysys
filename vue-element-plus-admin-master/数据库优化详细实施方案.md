# 数据库优化详细实施方案

## 一、当前数据库架构分析

### 1.1 数据库连接管理现状

**问题识别**：
```python
# 当前实现 - main.py line 1212
def get_db():
    connection = create_db_connection("USER_DB", config)
    if not connection:
        raise HTTPException(status_code=500, detail="无法连接到用户数据库")
    try:
        yield connection
    finally:
        connection.close()
```

**存在的问题**：
1. **连接池未充分利用**: `create_db_connection`函数虽然支持连接池，但`get_db`每次都创建新连接
2. **连接泄漏风险**: 异常情况下连接可能未正确关闭
3. **无连接超时控制**: 长时间运行的查询可能阻塞连接池
4. **连接池参数固定**: 无法根据负载动态调整

### 1.2 数据库查询性能问题

#### 方案1: 改进连接池管理

**实施文件**: `backend/database.py`

```python
import pymysql
from dbutils.pooled_db import PooledDB
from typing import Optional, Dict
import os
import logging

logger = logging.getLogger(__name__)

db_pools: Dict[str, PooledDB] = {}

def get_optimized_pool_config(section: str, config) -> dict:
    """根据系统资源和配置生成优化的连接池参数"""
    cpu_count = os.cpu_count() or 4
    
    # 根据CPU核心数动态调整连接池大小
    max_connections = min(cpu_count * 2 + 2, 20)
    min_cached = max(2, cpu_count // 2)
    max_cached = max(5, cpu_count)
    
    return {
        'host': config.get(section, 'host', fallback='localhost'),
        'port': config.getint(section, 'port', fallback=3306),
        'user': config.get(section, 'user', fallback='root'),
        'password': config.get(section, 'password', fallback='password'),
        'database': config.get(section, 'database', fallback='test'),
        'charset': config.get(section, 'charset', fallback='utf8mb4'),
        'maxconnections': max_connections,
        'mincached': min_cached,
        'maxcached': max_cached,
        'blocking': True,
        'ping': 1,  # 连接可用性检查
        'connect_timeout': 10,  # 连接超时
        'read_timeout': 30,     # 读超时
        'write_timeout': 30     # 写超时
    }

def create_optimized_db_pool(section: str, config) -> PooledDB:
    """创建优化的数据库连接池"""
    try:
        pool_config = get_optimized_pool_config(section, config)
        pool = PooledDB(pymysql, **pool_config)
        db_pools[section] = pool
        
        logger.info(f"创建连接池 {section}: max_connections={pool_config['maxconnections']}, "
                   f"min_cached={pool_config['mincached']}, max_cached={pool_config['maxcached']}")
        
        return pool
    except Exception as e:
        logger.error(f"创建 {section} 连接池失败: {e}")
        raise Exception(f"创建 {section} 连接池失败: {e}")

def get_connection_with_timeout(section: str, config, timeout: int = 5):
    """带超时控制的连接获取"""
    import time
    start_time = time.time()
    
    while time.time() - start_time < timeout:
        try:
            if section in db_pools:
                pool = db_pools[section]
                return pool.connection()
            else:
                return create_db_connection(section, config)
        except Exception as e:
            logger.warning(f"获取连接失败，重试中: {e}")
            time.sleep(0.1)
    
    raise Exception(f"获取数据库连接超时 ({timeout}秒)")
```

```python
def get_db():
    """FastAPI依赖项 - 使用连接池"""
    with get_db_context() as connection:
        yield connection
```

#### 优化1: 解决N+1查询问题

**优化前** (main.py line 1560-1572):

```python
async def get_users_list_optimized(
    pageIndex: Optional[int] = None,
    pageSize: Optional[int] = None,
    db = Depends(get_db)
):
    """优化后的用户列表查询 - 解决N+1问题"""
    try:
        with db.cursor(pymysql.cursors.DictCursor) as cursor:
            # 1. 查询总数
            cursor.execute("SELECT COUNT(*) as total FROM users")
            total = cursor.fetchone()['total']
            
            # 2. 分页查询用户基本信息
            if pageIndex and pageSize:
                offset = (pageIndex - 1) * pageSize
                cursor.execute("""
                    SELECT u.id, u.username, u.nickname, u.email, u.department_id, u.status,
                           d.name as department_name
                    FROM users u 
                    LEFT JOIN departments d ON u.department_id = d.id 
                    ORDER BY u.id
                    LIMIT %s OFFSET %s
                """, (pageSize, offset))
            else:
                cursor.execute("""
                    SELECT u.id, u.username, u.nickname, u.email, u.department_id, u.status,
                           d.name as department_name
                    FROM users u 
                    LEFT JOIN departments d ON u.department_id = d.id 
                    ORDER BY u.id
                """)
            
            users = cursor.fetchall()
            
            if not users:
                return {
                    "code": 200,
                    "message": "success",
                    "data": {
                        "list": [],
                        "total": total,
                        "pageIndex": pageIndex or 1,
                        "pageSize": pageSize or total
                    }
                }
            
            # 3. 批量查询所有用户的角色 - 一次查询解决N+1问题
            user_ids = [user['id'] for user in users]
            placeholders = ','.join(['%s'] * len(user_ids))
            
            cursor.execute(f"""
                SELECT ur.user_id, ur.role_id, r.name as role_name, r.code as role_code
                FROM user_roles ur
                JOIN roles r ON ur.role_id = r.id
                WHERE ur.user_id IN ({placeholders})
            """, user_ids)
            
            user_roles_data = cursor.fetchall()
            
            # 4. 构建用户角色映射
            user_roles_map = {}
            for role_data in user_roles_data:
                user_id = role_data['user_id']
                if user_id not in user_roles_map:
                    user_roles_map[user_id] = []
                user_roles_map[user_id].append({
                    'role_id': role_data['role_id'],
                    'role_name': role_data['role_name'],
                    'role_code': role_data['role_code']
                })
            
            # 5. 组装最终结果
            for user in users:
                user['roles'] = user_roles_map.get(user['id'], [])
            
            return {
                "code": 200,
                "message": "success",
                "data": {
                    "list": users,
                    "total": total,
                    "pageIndex": pageIndex or 1,
                    "pageSize": pageSize or total
                }
            }
            
    except Exception as e:
        logger.error(f"获取用户列表失败: {e}")
        return {"code": 500, "message": "获取用户列表失败"}
```

**性能提升**：
- 查询次数: 从 N+1 次减少到 3 次
- 100个用户: 从101次查询减少到3次查询
- 响应时间: 预计提升 90%+

### 2.4 查询缓存实现

#### 方案: 简单内存缓存

**实施文件**: `backend/cache.py`

```python
import time
from typing import Any, Optional, Dict
import logging

logger = logging.getLogger(__name__)

class SimpleCache:
    """简单的内存缓存实现"""
    
    def __init__(self, default_ttl: int = 300):
        self.cache: Dict[str, Dict[str, Any]] = {}
        self.default_ttl = default_ttl
    
    def get(self, key: str) -> Optional[Any]:
        """获取缓存值"""
        if key not in self.cache:
            return None
        
        item = self.cache[key]
        if time.time() > item['expires_at']:
            del self.cache[key]
            return None
        
        return item['value']
    
    def set(self, key: str, value: Any, ttl: Optional[int] = None):
        """设置缓存值"""
        expires_at = time.time() + (ttl or self.default_ttl)
        self.cache[key] = {
            'value': value,
            'expires_at': expires_at
        }
    
    def delete(self, key: str):
        """删除缓存值"""
        if key in self.cache:
            del self.cache[key]
    
    def clear_pattern(self, pattern: str):
        """清除匹配模式的缓存"""
        keys_to_delete = [key for key in self.cache.keys() if key.startswith(pattern)]
        for key in keys_to_delete:
            del self.cache[key]
    
    def get_stats(self):
        """获取缓存统计信息"""
        current_time = time.time()
        valid_items = sum(1 for item in self.cache.values() if item['expires_at'] > current_time)
        return {
            'total_items': len(self.cache),
            'valid_items': valid_items,
            'expired_items': len(self.cache) - valid_items
        }

# 全局缓存实例
cache = SimpleCache(default_ttl=300)  # 5分钟默认TTL

def cache_key(*args, **kwargs):
    """生成缓存键"""
    key_parts = [str(arg) for arg in args]
    key_parts.extend(f"{k}={v}" for k, v in sorted(kwargs.items()))
    return ":".join(key_parts)
```

```python
import time
import logging
from functools import wraps

logger = logging.getLogger(__name__)

def monitor_query_performance(func):
    """查询性能监控装饰器"""
    @wraps(func)
    async def wrapper(*args, **kwargs):
        start_time = time.time()
        try:
            result = await func(*args, **kwargs)
            execution_time = time.time() - start_time
            
            # 记录慢查询 (>1秒)
            if execution_time > 1.0:
                logger.warning(f"慢查询: {func.__name__} 耗时 {execution_time:.2f}秒")
            else:
                logger.debug(f"查询: {func.__name__} 耗时 {execution_time:.3f}秒")
            
            return result
        except Exception as e:
            execution_time = time.time() - start_time
            logger.error(f"查询异常: {func.__name__} 耗时 {execution_time:.2f}秒, 错误: {e}")
            raise
    
    return wrapper
```

## 六、风险评估与缓解

### 6.1 主要风险

1. **索引创建锁表**
   - 风险: 大表创建索引可能锁表
   - 缓解: 在低峰期执行，使用ONLINE DDL

2. **查询优化引入bug**
   - 风险: 优化后的查询可能逻辑错误
   - 缓解: 充分测试，保留原实现作为备份

3. **缓存一致性问题**
   - 风险: 缓存数据与数据库不一致
   - 缓解: 合理设置TTL，及时清除缓存

### 6.2 监控指标

- 查询响应时间

- 慢查询数量
- 缓存命中率
